        function rollDice() {

            if (gameComplete) return;

            

            const dice = Math.floor(Math.random() * 6) + 1;

            document.getElementById('rollDice').textContent = dice;

            

            setTimeout(() => {

                document.getElementById('rollDice').textContent = '🎲';

                movePlayer(dice);

            }, 1000);

        }


        function movePlayer(steps) {

            currentPosition += steps;

            

            if (currentPosition >= 13) {

                currentPosition = 13;

                gameComplete = true;

                alert('Congratulations! You\'ve completed your journey of Walking in Good Relations. Remember, this is a lifelong journey of learning and unlearning.');

            }

            

            updatePlayerPosition();

            

            if (currentPosition > 0 && currentPosition <= 13) {

                const currentSquare = gameSquares[currentPosition - 1];

                setTimeout(() => showSquareDetails(currentSquare), 500);

            }

        }


        function updatePlayerPosition() {

            const positionText = currentPosition === 0 ? 'Start' : 

                               currentPosition === 13 ? 'Journey Complete!' : 

                               `Square ${currentPosition}`;

            document.getElementById('playerPosition').textContent = `Position: ${positionText}`;

            

            // Highlight current square

            document.querySelectorAll('.square').forEach((square, index) => {

                square.style.border = index === currentPosition - 1 ? '3px solid #ca3941' : '';

            });

        }

        function flipCard() {

            const definitionSide = document.querySelector('.definition-side');

            const discussionSide = document.querySelector('.discussion-side');

            const card = document.querySelector('.vocab-card');

            

            if (currentCardSide === 'definition') {

                definitionSide.classList.remove('active');

                discussionSide.classList.add('active');

                card.classList.add('flipped');

                currentCardSide = 'discussion';

            } else {

                discussionSide.classList.remove('active');

                definitionSide.classList.add('active');

                card.classList.remove('flipped');

                currentCardSide = 'definition';

            }

        }
